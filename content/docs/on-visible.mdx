---
title: onVisible()
description: Trigger animations when elements scroll into the viewport.
---

Fires a callback when an element enters the viewport. Wraps `IntersectionObserver`. Returns a cleanup function.

```ts
import { onVisible, enter } from '@tweens/tweens'

const cleanup = onVisible(el, (el) =>
  enter(el, { y: 20, opacity: 0 }, 'snappy')
)

// Disconnect on unmount
cleanup()
```

## In React

```tsx
import { useEffect, useRef } from 'react'
import { onVisible, enter } from '@tweens/tweens'

export function FadeInSection({ children }) {
  const ref = useRef<HTMLDivElement>(null)

  useEffect(() => {
    if (!ref.current) return
    return onVisible(
      ref.current,
      (el) => enter(el, { y: 24, opacity: 0 }, 'gentle'),
      { threshold: 0.15 }
    )
  }, [])

  return (
    <div ref={ref} className="opacity-0">
      {children}
    </div>
  )
}
```

## Multiple elements

```ts
// NodeList
onVisible(
  document.querySelectorAll('.card'),
  (el) => enter(el, { y: 16, opacity: 0 }, 'snappy'),
  { threshold: 0.1 }
)

// Array of refs
onVisible(
  itemRefs.current,
  (el) => enter(el, { scale: 0.95, opacity: 0 }, 'gentle')
)
```

## Signature

```ts
onVisible(
  target: HTMLElement | HTMLElement[] | NodeListOf<Element>,
  callback: (el: HTMLElement) => void,
  options?: {
    threshold?: number    // 0–1 visibility ratio, default 0.1
    once?: boolean        // fire once then unobserve, default true
    rootMargin?: string   // IntersectionObserver rootMargin, default '0px'
  },
): () => void
```

## Options

| Option | Default | Description |
|--------|---------|-------------|
| `threshold` | `0.1` | How much of the element must be visible (0–1) |
| `once` | `true` | Fire only once per element |
| `rootMargin` | `'0px'` | Expand or shrink the detection zone |
